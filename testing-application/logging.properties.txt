## log4j
# Apache's log4j is used to provide system logging.
##

# Levels in order of increasing verbosity and log size, and is decreasing
# order of severity are:
#		OFF - no logging
#		ERROR - something serious has happened that might affect any 
#					subsequent actions.
#		WARN - something has not been set up properly, or needs 
#					attention.  Although subsequent action might function 
#					correctly, they might not function as expected.
#		INFO - information to the developer about something that has 
#					taken place.
# 		DEBUG - detailed information for the developer.
#
#
# the root level is the overall level of logging that will be generated, 
# unless specifically overriden.  The specified appenders list the ways that
# the logs will be written.
#
log4j.rootCategory=DEBUG, Console, File, XmlFile
#log4j.logger.org.nakedobjects=DEBUG, Console
#log4j.logger.org.nakedobjects=DEBUG, File


#
# We can specify that messages for specific classes or packages are logged 
# at a different level.
#
# this line requests that the ObjectLoaderImpl logs are limited to INFO or above
log4j.logger.org.nakedobjects.objects.defaults.ObjectLoaderImpl=INFO

# this one restricts all viewer logging to warning and errors
log4j.logger.org.nakedobjects.viewer=WARN



# The console appender outputs to the console - useful only when a console
# is visible
log4j.appender.Console=org.apache.log4j.ConsoleAppender
log4j.appender.Console.target=System.out
log4j.appender.Console.layout=org.apache.log4j.PatternLayout
log4j.appender.Console.layout.ConversionPattern=%-5r [%-20c{1} %-10t %-5p]  %m%n

# This file appender logs to a file called nakedobjects.log (in the working 
# directory), does not append to an existing file (creates a new file each
# time the application is run), and writes each entry in a format as defined
# by the pattern
log4j.appender.File=org.apache.log4j.FileAppender
log4j.appender.File.file=nakedobjects.log
log4j.appender.File.append=false
log4j.appender.File.layout=org.apache.log4j.PatternLayout
log4j.appender.File.layout.ConversionPattern=%d [%-20c{1}] %-5p  %m%n

# This appender also creates a file but writes out each event as XML block.
log4j.appender.XmlFile=org.apache.log4j.FileAppender
log4j.appender.XmlFile.file=nakedobjects.log.xml
log4j.appender.XmlFile.append=false
log4j.appender.XmlFile.layout=org.apache.log4j.xml.XMLLayout










# configuration for CLIENT
#   - Console output is limited to WARN, and limited to message only
#   - a configuration for XML log is provided but not turned on
log4j.rootCategory=DEBUG, Popup, Remote

log4j.appender.XmlFile=org.apache.log4j.FileAppender
log4j.appender.XmlFile.file=nakedobjects.log.xml
log4j.appender.XmlFile.append=false
log4j.appender.XmlFile.layout=org.apache.log4j.xml.XMLLayout

log4j.appender.Remote=org.nakedobjects.utility.WebSnapshotAppender
#log4j.appender.Remote.subject=Error from NO (msg speced in logging confilog4j.appender.Remote.subject=Naked Objects Client
#log4j.appender.Remote.layout=org.apache.log4j.PatternLayout
#log4j.appender.Remote.layout.ConversionPattern=%-5r [%-20c{1} %-10t %-5p]  %m  -- %l%n
log4j.appender.Remote.layout=org.apache.log4j.HTMLLayout
log4j.appender.Remote.layout.locationInfo=false
log4j.appender.Remote.proxyAddress=<proxy server>
#log4j.appender.Remote.proxyPort=8080

log4j.appender.Popup=org.nakedobjects.utility.PopupSnapshotAppender
log4j.appender.Popup.layout=org.apache.log4j.PatternLayout
log4j.appender.Popup.layout.ConversionPattern=%-5r [%-20c{1} %-10t %-5p]  %m%n
#log4j.appender.Popup.layout.locationInfo=true



# configuration for SERVER
#   - No console output, but simple output to a file, which can be tailed; 
#		 limited to WARN; and shows message only
#   - ignore naked object repository events
#   - a configuration for XML log is provided but not turned on
		 
log4j.rootCategory=DEBUG, Console, Remote

log4j.org.nakedobject.NakedObjectsByThread=OFF


log4j.appender.Console=org.apache.log4j.FileAppender
log4j.appender.Console.file=console.log
log4j.appender.Console.append=false
log4j.appender.Console.threshold=WARN
log4j.appender.Console.layout=org.apache.log4j.SimpleLayout

log4j.appender.XmlFile.file=nakedobjects.log.xml
log4j.appender.XmlFile.append=false
log4j.appender.XmlFile.layout=org.apache.log4j.xml.XMLLayout

log4j.appender.Remote=org.nakedobjects.utility.WebSnapshotAppender

log4j.appender.Remote.layout=org.apache.log4j.HTMLLayout
